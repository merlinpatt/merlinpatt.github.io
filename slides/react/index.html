Possible Solution 

              
                class Message extends React.Component {
                  render () {
                    const {avatarUrl, name, text, timestamp} = this.props;
                    const time = timestamp.toTimeString().split(' ')[0];

                    styles.timeMerged = styles.timeNotExpanded;
                    styles.timeMerged.color = styles.time.color;

                    return (
                      <div>
                        {expanded &&
                          <img src={avatarUrl} styles={styles.image} />
                        }
                        <div style={styles.rightSide}>
                          {! expanded &&
                            <span style={styles.timeMerged}>
                              {time}
                            </span>
                          }
                          <span style={styles.name}>
                            {name}
                          </span>
                          {expanded &&
                            <span style={styles.time}>
                              {time}
                            </span>
                          }
                          <p style={expanded ? styles.text : styles.textNotExpanded}>
                            {text}
                          </p>
                        </div>
                      </div>
                    );
                  }
                }

                const styles = {
                  // original styles

                    display: 'inline',
                  },
                  timeNotExpanded: {
                    fontSize: '12px',
